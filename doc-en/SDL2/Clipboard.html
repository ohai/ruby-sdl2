static VALUE Clipboard_s_text(VALUE self)
{
    if (SDL_HasClipboardText()) {
        char* text = SDL_GetClipboardText();
        VALUE str;
        if (!text)
            SDL_ERROR();
        str = utf8str_new_cstr(text);
        SDL_free(text);
        return str;
    } else {
        return Qnil;
    }
}
